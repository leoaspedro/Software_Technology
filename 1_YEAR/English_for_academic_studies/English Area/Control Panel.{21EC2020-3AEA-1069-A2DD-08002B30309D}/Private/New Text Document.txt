    Private Sub CalcularToolStripMenuItem_Click(sender As System.Object, e As 

System.EventArgs) Handles CalcularToolStripMenuItem.Click

        Dim base, altura, area, perimetro As Single

        base = Val(InputBox("Digite o valor da base"))

        altura = Val(InputBox("digite o valor da altura"))

        area = base * altura

        perimetro = 2 * base + 2 * altura

        area = MsgBox(" A área é igual a:    " & area)

        perimetro = MsgBox(" O perímetro é igual a:    " & perimetro)

    End Sub
